TRIGGER "SAJET".G_ERP_RT_ITEM_TRIGGER
AFTER DELETE OR INSERT OR UPDATE
ON SAJET.G_ERP_RT_ITEM REFERENCING NEW AS NEW OLD AS OLD
FOR EACH ROW
DECLARE
NEWLOTNO SAJET.G_IQC_LOT.LOT_NO%TYPE;
V_VENDOR SAJET.G_ERP_RTNO.VENDOR_ID%TYPE;
V_PO SAJET.G_ERP_RTNO.PO_NO%TYPE;
V_LOTNO SAJET.G_IQC_LOT.LOT_NO%TYPE;
V_INSP_LOT SAJET.G_IQC_LOT.INSP_LOT%TYPE;
V_RT_NO SAJET.G_ERP_RTNO.RT_NO%TYPE;
V_TARGETTIME DATE;
BEGIN
 IF INSERTING OR UPDATING THEN  
 BEGIN
--  SELECT WAREHOUSE_ID INTO V_WAREHOUSE
--  FROM SAJET.G_IQC_SP_WAREHOUSE WHERE WAREHOUSE_ID = :NEW.LOCATION AND ROWNUM = 1;
    
  V_INSP_LOT := 'Y';
 EXCEPTION
    WHEN NO_DATA_FOUND THEN 
       V_INSP_LOT := 'N';
 END;   
 
-- V_INSP_LOT := 'Y';
  
 BEGIN 
  SELECT SUBSTR(RT_NO,1,3) INTO V_RT_NO
    FROM SAJET.G_ERP_RTNO WHERE RT_ID = :NEW.RT_ID
     AND ROWNUM = 1;
  IF LENGTH(V_RT_NO) >=3 THEN
    V_RT_NO := SUBSTR(V_RT_NO,1,3);   
  END IF;   
  
 EXCEPTION
    WHEN NO_DATA_FOUND THEN 
       GOTO ENDP;
 END;
 END IF;  
  
  
  IF INSERTING THEN
    V_TARGETTIME:= SAJET.GET_IQC_LOT_TARGETTIME(NEWLOTNO);
    SELECT VENDOR_ID, PO_NO INTO V_VENDOR, V_PO
      FROM SAJET.G_ERP_RTNO WHERE RT_ID = :NEW.RT_ID AND ROWNUM = 1; 
    SELECT RT_NO || '-' || LPAD(:NEW.RT_SEQ, 4, '0') INTO NEWLOTNO
      FROM SAJET.G_ERP_RTNO WHERE RT_ID = :NEW.RT_ID;
    INSERT INTO SAJET.G_IQC_LOT
    (LOT_NO, EMP_ID, RT_ID, RT_SEQ, PART_ID, VERSION, LOT_SIZE, VENDOR_ID, PO_NO,START_TIME,URGENT_TYPE, INSP_LOT) 
    VALUES (NEWLOTNO, :NEW.UPDATE_USERID, :NEW.RT_ID, :NEW.RT_SEQ, :NEW.PART_ID, 
    :NEW.PART_VERSION, :NEW.INCOMING_QTY, V_VENDOR, :NEW.PO_NO,:NEW.INCOMING_TIME,:NEW.URGENT_TYPE, V_INSP_LOT);
     BEGIN
      SELECT SAJET.GET_IQC_LOT_TARGETTIME(NEWLOTNO) INTO V_TARGETTIME FROM DUAL;
      UPDATE SAJET.G_IQC_LOT SET TARGET_TIME = V_TARGETTIME WHERE LOT_NO = NEWLOTNO;
    EXCEPTION
      WHEN OTHERS THEN 
         NULL;          
    END; 
  ELSIF UPDATING THEN
    UPDATE SAJET.G_IQC_LOT
   SET LOT_SIZE = :NEW.INCOMING_QTY, PART_ID = :NEW.PART_ID, VERSION = :NEW.PART_VERSION,START_TIME =:NEW.INCOMING_TIME
      ,PO_NO = :NEW.PO_NO,URGENT_TYPE =:NEW.URGENT_TYPE, INSP_LOT = V_INSP_LOT
     WHERE RT_ID = :NEW.RT_ID AND RT_SEQ = :NEW.RT_SEQ;   

     --add by rita 2014/02/12 for 批號目標日計算
    IF :NEW.URGENT_TYPE <>:OLD.URGENT_TYPE THEN
      BEGIN
        SELECT LOT_NO INTO V_LOTNO FROM SAJET.G_IQC_LOT WHERE RT_ID = :NEW.RT_ID AND RT_SEQ = :NEW.RT_SEQ;
        SELECT SAJET.GET_IQC_LOT_TARGETTIME(V_LOTNO) INTO V_TARGETTIME FROM DUAL;
        UPDATE SAJET.G_IQC_LOT SET TARGET_TIME = V_TARGETTIME WHERE LOT_NO = V_LOTNO;
      EXCEPTION
         WHEN OTHERS THEN 
            NULL;               
      END;              
    END IF;       
  ELSIF DELETING THEN
    BEGIN
      SELECT LOT_NO INTO V_LOTNO FROM SAJET.G_IQC_LOT WHERE RT_ID = :OLD.RT_ID AND RT_SEQ = :OLD.RT_SEQ;
      DELETE FROM SAJET.G_IQC_DETAIL WHERE LOT_NO = V_LOTNO;
      DELETE FROM SAJET.G_IQC_TEST_ITEM_VALUE WHERE LOT_NO = V_LOTNO;
      DELETE FROM SAJET.G_IQC_TEST_ITEM WHERE LOT_NO = V_LOTNO;
      DELETE FROM SAJET.G_IQC_TEST_TYPE WHERE LOT_NO=V_LOTNO;
      DELETE FROM SAJET.G_IQC_TEST_TYPE_DEFECT WHERE LOT_NO = V_LOTNO;
      DELETE FROM SAJET.G_IQC_TOOLING WHERE LOT_NO = V_LOTNO;
      DELETE FROM SAJET.G_IQC_VENDOR_PART WHERE LOT_NO = V_LOTNO;
      DELETE FROM SAJET.G_IQC_DEFECT WHERE LOT_NO = V_LOTNO;
      DELETE FROM SAJET.G_IQC_LOT WHERE LOT_NO = V_LOTNO;
    EXCEPTION
        WHEN NO_DATA_FOUND THEN 
              NULL;
    END;      
  END IF;
  
   <<ENDP>>
     NULL;
END G_ERP_RT_ITEM_TRIGGER;
